# 最长回文子序列 dp[i][j]表示s[i:j]的最长回文子序列长度
```javascript
var longestPalindromeSubseq = function (s) {
  const l = s.length,
    dp = new Array(l);
  for (let i = 0; i < l; i++) {
    dp[i] = new Array(l).fill(0);
    dp[i][i] = 1;  // 赋值1很重要
  }

  for (let i = l - 1; i >= 0; i--) {
    for (let j = i + 1; j < l; j++) {
      if (s[i] === s[j]) {
        dp[i][j] = dp[i + 1][j - 1] + 2;
      } else {
        dp[i][j] = Math.max(dp[i + 1][j], dp[i][j - 1]);
      }
    }
  }
  return dp[0][l - 1];
};
console.log(longestPalindromeSubseq("bbbab"));
```

# 最长回文子串 dp[i][j]===1表示s[i:j]是回文串
```javascript
var longestPalindrome = function (s) {
  const l = s.length,
    dp = new Array(l);
  let res = s[0];
  for (let i = 0; i < l; i++) {
    dp[i] = new Array(l).fill(0);
    dp[i][i] = 1;
  }

  for (let i = l - 1; i >= 0; i--) {
    for (let j = i + 1; j < l; j++) {
      if (s[i] === s[j] && (j - i === 1 || dp[i + 1][j - 1] === 1)) {
        dp[i][j] = 1;
        if (j - i + 1 > res.length) {
          res = s.substring(i, j+1);
        }
      }
    }
  }
  return res;
};
```


# 全排列
```javascript
var permute = function(nums) {
    const l = nums.length;
    const res = [];
    function dfs(ed, able) {
        if (ed.length===l) {
            res.push([...ed]);
            return;
        }

        for (let i=0; i<able.length; i++) {
            if (ed.indexOf(able[i])>-1) continue; // 如果有相同的元素 则跳过
            ed.push(able[i]);
            dfs(ed, able);
            ed.pop();
        }
    }

    dfs([], nums);
    return res;
};
```

# 无重复全排列
```javascript
var permute = function (nums) {
  const l = nums.length;
  const res = [];
  nums.sort((a, b)=>a-b);  // 无重复的一定需要排序
  function dfs(ed, able) {
    if (ed.length === l) {
      res.push([...ed]);
      return;
    }

    for (let i = 0; i < able.length; i++) {
      if (i > 0 && able[i] === able[i - 1]) {
        continue;
      }
      ed.push(able[i]);
      let tmp = [...able]; // 新建一个辅助数组
      tmp.splice(i, 1); // 去除已经选取的元素
      dfs(ed, tmp);
      ed.pop();
    }
  }

  dfs([], nums);
  return res;
};

console.log(permute([1, 1, 2]));
```

# 无重复最长子串 双指针start, end
```javascript
var lengthOfLongestSubstring = function (s) {
  let m = new Map(),
    start = 0,
    end = 1,
    l = s.length,
    res = 1;
  if (s.length === 0) return 0;
  m.set(s[0], 0);
  while (end < l) {
    let char = s[end];
    if (!m.has(char) || m.get(char) < start) {
      res = Math.max(res, end - start + 1);
      m.set(char, end);
      end++;
    } else {
      start = m.get(char) + 1;
    }
  }
  return res;
};
```

```javascript
var isSymmetric = function (root) {
  function isTowTree(s, t) {
    if (s === null && t === null) return true;
    if (s === null || t === null) return false;
    return s.val === t.val && isTowTree(s.left, t.right) && isTowTree(s.right, t.left);
  }
  return isTowTree(root.left, root.right);
};
```

## 二叉树的最小深度
```javascript
var minDepth = function (root) {
  if (root === null) return 0;
  let l = minDepth(root.left),
    r = minDepth(root.right);
  // 必须是叶子节点才是目标节点
  if (root.left === null && root.right !== null) {  
    return 1 + r;
  }
  if (root.right === null && root.left !== null) {
    return 1 + l;
  }
  return 1 + Math.min(l, r);
};
```

## 平衡二叉树
```javascript
var isBalanced = function (root) {
  function getHeight(root) {
    if (root === null) return 0;
    let l = getHeight(root.left),
      r = getHeight(root.right);
    return Math.max(l, r) + 1;
  }

  if (root === null) return true;
  return Math.abs(getHeight(root.left) - getHeight(root.right)) < 2 && isBalanced(root.left) && isBalanced(root.right);
};
```

```javascript
var binaryTreePaths = function (root) {

    if (root.left) {
      backTrace(root.left, path)

    if (root.right) {
      backTrace(root.right, path)
    }
  }
```

```javascript
var pathSum = function (root, targetSum) {
  const res = [];
  function backTrace(root, targetSum, path) {
    if (root === null) return [];
    if (root.left === null && root.right === null && root.val === targetSum) {
      res.push([...path, root.val]);
    }

    path.push(root.val)
    if (root.left !== null) {
      backTrace(root.left, targetSum - root.val, path);
      path.pop();
    }
    if (root.right !== null) {
      backTrace(root.right, targetSum - root.val, path);
      path.pop();
    }
  }

  backTrace(root, targetSum, []);
  return res;
};
```

# dfs
## 矩阵中的路径
```javascript
var exist = function (board, word) {
  const row = board.length,
    col = board[0].length;

  function dfs(i, j, k) {
    if (i < 0 || i >= row || j < 0 || j >= col || word[k] !== board[i][j]) {
      return false;
    }
    if (k === word.length - 1) return true;

    board[i][j] = '';
    res = dfs(i - 1, j, k + 1) || dfs(i + 1, j, k + 1) || dfs(i, j - 1, k + 1) || dfs(i, j + 1, k + 1)
    board[i][j] = word[k];
    return res;
  }

  for (let i = 0; i < row; i++) {
    for (let j = 0; j < col; j++) {
      if (dfs(i, j, 0)) return true;
    }
  }

  return false;
};
```

# 单调栈
```javascript
输入: temperatures = [73,74,75,71,69,72,76,73]
输出: [1,1,4,2,1,1,0,0]
var dailyTemperatures = function (temperatures) {
  const stack = [],
    l = temperatures.length,
    res = new Array(l).fill(0);

  for (let i = 0; i < l; i++) {
    while (stack.length > 0 && temperatures[i] > temperatures[stack[stack.length - 1]]) {
      let prev = stack.pop(); // 往前找第一个比当前温度小的索引prev
      res[prev] = i - prev;   // 长度就为当前索引i减去prev
    }
    stack.push(i);
  }
  return res;
};
```

# 快速排序
```javascript
function getIndex(nums, l, r) {
  const tmp = nums[l];
  while (l < r) {
    while (l < r && nums[r] >= tmp) {
      r--;
    }
    nums[l] = nums[r];
    while (l < r && nums[l] < tmp) {
      l++;
    }
    nums[r] = nums[l];
  }
  nums[l] = tmp;
  return l;
}

function quickSort(nums, l, r) {
  if (l < r) {
    const mid = getIndex(nums, l, r);
    quickSort(nums, l, mid - 1);
    quickSort(nums, mid + 1, r);
  }
}

const nums = [5, 4, 3, 2, 1];

console.log(nums);
```

## 把数组的‘3’放到后面的位置
```javascript
const nums = [0, 1, 0, 3, 3, 3, 12, 3, 1, 6, 3];
function f(nums) {
  let count = 0; // 记录在当前位置‘3’出现了多少次
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] === 3) {
      // 如果当前数是3，那就计数
      count++;
    } else if (count > 0) {
      // 如果当前数不是3，并且前面有3
      let n = i;
      nums[i - count] = nums[i];
      nums[n] = 3;
    }
  }
  return nums;
}
console.log(f(nums));
```
